      requirejs.config({
        baseUrl: '../../../'
      });
      require(['mds', 'jquery'], function() {

        $('a').on('click tap', function (event) {
          if($(event.target).attr("rel") !== 'profile-popover') {
            event.preventDefault();
          }
        });

        $('[rel="profile-popover"]').popover({
          html: true,
          content: function() {
            return $('.profile-card').html();
          },
          placement: 'top',
          container: 'body'
        }).on('shown', function() {
          $('[rel=tooltip]').tooltip({ container: 'body' });
        });

        $('[rel="profile-popover"]').on('click tap', function (event) {
          // prevent the follow of the href in the parent anchor
          event.preventDefault();
          $('[rel="profile-popover"]').not(this).popover('hide');
        });

        $(document).on('click tap', function (event) {
          if( ( $(event.target).parents('.popover').length === 0 &&
              $(event.target).attr('rel') !== 'profile-popover') ||
              $(event.target).parents('.close').length > 0 ||
              $(event.target).hasClass('close')) {
            // close the popover
            $('[rel="profile-popover"]').popover('hide');
          }
        });

        $(window).on('resize', function (event) {
          $('[rel="profile-popover"]').popover('hide');
        });


      });
  </script>
